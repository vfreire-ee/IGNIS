BEGIN TRANSACTION;
CREATE TABLE IF NOT EXISTS "PID" (
	"PID_ID"	INTEGER NOT NULL UNIQUE,
	"Nome"	INTEGER NOT NULL UNIQUE,
	"Kp"	NUMERIC NOT NULL,
	"Ki"	NUMERIC NOT NULL,
	"Kd"	NUMERIC NOT NULL,
	PRIMARY KEY("PID_ID" AUTOINCREMENT)
);
CREATE TABLE IF NOT EXISTS "Setor" (
	"Setor_ID"	INTEGER NOT NULL UNIQUE,
	"Nome"	TEXT NOT NULL UNIQUE,
	PRIMARY KEY("Setor_ID" AUTOINCREMENT)
);
CREATE TABLE IF NOT EXISTS "Historico" (
	"Historico_ID"	INTEGER NOT NULL UNIQUE,
	"Temperatura"	NUMERIC NOT NULL,
	"Pressao"	NUMERIC NOT NULL,
	"Data"	NUMERIC NOT NULL,
	"Horario"	NUMERIC NOT NULL,
	"Setor_ID"	INTEGER NOT NULL,
	PRIMARY KEY("Historico_ID" AUTOINCREMENT),
	FOREIGN KEY("Setor_ID") REFERENCES "Setor"("Setor_ID")
);
CREATE TABLE IF NOT EXISTS "Relatorios" (
	"Relatorio_Id"	INTEGER NOT NULL UNIQUE,
	"Titulo"	TEXT NOT NULL,
	"Descricao"	TEXT NOT NULL,
	"Data"	NUMERIC NOT NULL,
	"Horario"	NUMERIC NOT NULL,
	"Link"	TEXT NOT NULL,
	"Engenheiro_ID"	INTEGER NOT NULL,
	"Tecnico_ID"	INTEGER NOT NULL,
	PRIMARY KEY("Relatorio_Id" AUTOINCREMENT),
	FOREIGN KEY("Engenheiro_ID") REFERENCES "Empregado"("Empregado_ID"),
	FOREIGN KEY("Tecnico_ID") REFERENCES "Empregado"("Empregado_ID")
);
CREATE TABLE IF NOT EXISTS "Caldeira" (
	"Caldeira_ID"	INTEGER NOT NULL UNIQUE,
	"Nome"	TEXT NOT NULL UNIQUE,
	"Setor_ID"	INTEGER NOT NULL,
	"Status"	INTEGER NOT NULL,
	"Setpoint"	REAL NOT NULL,
	"Data"	NUMERIC NOT NULL,
	"Horario"	NUMERIC NOT NULL,
	"PID_ID"	INTEGER NOT NULL,
	PRIMARY KEY("Caldeira_ID" AUTOINCREMENT),
	FOREIGN KEY("Setor_ID") REFERENCES "Setor"("Setor_ID"),
	FOREIGN KEY("PID_ID") REFERENCES "PID"("PID_ID")
);
CREATE TABLE IF NOT EXISTS "Coluna_destilacao" (
	"Coluna_ID"	INTEGER NOT NULL UNIQUE,
	"Nome"	TEXT NOT NULL UNIQUE,
	"Setor_ID"	INTEGER NOT NULL,
	"Status"	INTEGER NOT NULL,
	"Setpoint"	NUMERIC NOT NULL,
	"Data"	NUMERIC NOT NULL,
	"Horario"	NUMERIC NOT NULL,
	"PID_ID"	INTEGER NOT NULL,
	PRIMARY KEY("Coluna_ID" AUTOINCREMENT),
	FOREIGN KEY("Setor_ID") REFERENCES "Setor"("Setor_ID"),
	FOREIGN KEY("PID_ID") REFERENCES "PID"("PID_ID")
);
CREATE TABLE IF NOT EXISTS "Empregado" (
	"Empregado_ID"	INTEGER NOT NULL UNIQUE,
	"Nome"	TEXT NOT NULL,
	"CPF"	TEXT NOT NULL UNIQUE,
	"Sexo"	TEXT,
	"Data de Nascimento"	TEXT NOT NULL,
	"Telefone"	INTEGER,
	"Celular"	NUMERIC,
	"Cargo"	TEXT NOT NULL,
	"Email"	TEXT NOT NULL,
	"Senha"	TEXT NOT NULL,
	"Rua"	TEXT NOT NULL,
	"NÃºmero"	INTEGER,
	"Bairro"	TEXT NOT NULL,
	"Cidade"	TEXT NOT NULL,
	"Estado"	TEXT NOT NULL,
	"Pais"	TEXT NOT NULL,
	"Complemento"	TEXT,
	"CEP"	NUMERIC NOT NULL,
	"Setor_ID"	INTEGER NOT NULL,
	"Supervisor_ID"	INTEGER,
	PRIMARY KEY("Empregado_ID" AUTOINCREMENT),
	FOREIGN KEY("Supervisor_ID") REFERENCES "Empregado"("Empregado_ID"),
	FOREIGN KEY("Setor_ID") REFERENCES "Setor"("Setor_ID")
);
